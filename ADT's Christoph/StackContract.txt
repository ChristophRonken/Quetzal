
ADT Stack:

createStack()
// Creates Stack with an empty array
// precondition: self.stack == None
// postcondition: an array for the stack is created

destroyStack()
// Destroys Stack
// precondition:
// postcondition:   the array is deleted completely

isEmpty(): boolean
// Checks if Stack is empty
// precondition:
// postcondition:   return True: Stack is empty
                    return False: Stack is not empty

push(newitem: StackItemType): boolean
// Adds newitem to Stack
// precondition:
// postcondition:   return True: element added

pop(): boolean
// Removes top from Stack
// precondition:
// postcondition:   return True: top removed

popDisplay(): boolean
// Removes top from Stack and saves value
// precondition:
// postcondition:   return True: top removed
                    self.stacktop holds value of removed item

getTop(): boolean
// Saves top value of Stack
// precondition:
// postcondition:   return True: top saved
